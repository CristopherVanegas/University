db.createCollection("Job_History", { validator: { $jsonSchema: { bsonType: "object", required: [ "employee_id", "start_date", "end_date", "job_id", "department_id" ], properties: { employee_id: { bsonType: "string" }, start_date: { bsonType: "date" }, end_date: { bsonType: "date" }, job_id: { bsonType: "string" }, department_id: { bsonType: "string" } } } } })

  >  Cassandra

CREATE KEYSPACE job_history_keyspace
WITH REPLICATION = {'class': 'SimpleStrategy', 'replication_factor': 1};

CREATE TABLE job_history (
    EMPLOYEE_ID INT,
    START_DATE TEXT,
    END_DATE TEXT,
    JOB_ID TEXT,
    DEPARTMENT_ID INT,
    PRIMARY KEY (EMPLOYEE_ID)
);

INSERT INTO job_history (EMPLOYEE_ID, START_DATE, END_DATE, JOB_ID, DEPARTMENT_ID)
VALUES (102, '13-Jan-01', '24-Jul-06', 'IT_PROG', 60);



CREATE KEYSPACE jobs_keyspace
WITH REPLICATION = {'class': 'SimpleStrategy', 'replication_factor': 1};

CREATE TABLE jobs (
    JOB_ID TEXT PRIMARY KEY,
    JOB_TITLE TEXT,
    MIN_SALARY INT,
    MAX_SALARY INT
);

INSERT INTO jobs (JOB_ID, JOB_TITLE, MIN_SALARY, MAX_SALARY)
